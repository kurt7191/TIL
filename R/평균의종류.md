```R
#평균의 종류

#산술평균
x <- sample(1:10, 8, replace = T)
mean(x)

#조화평균
#서울 -> 부산 : 시속 100km/h
#부산 -> 대전 : 시속 80km/h 
#대전 -> 강릉 : 시속 90km/h

y <- c(100, 80, 90)
length(y) / sum(1/y)

#기하평균
#경제성장률
#2000년 5%, 2001년7%
z <- c(1.05, 1.07)
prod(z)^(1/length(z))

#가중평균
x <- c(2,5,7,9,10)
wt <- c(1.2,1.6,1.1,1.8,1.5)
weighted.mean(x, wt)

#절사평균

mean(x, trim = 0.2) #0.2면 양쪽 다해서 20퍼 자르고 평균.

```
